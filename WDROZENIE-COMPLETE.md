```bash

# - Kup domenę: pt24.pro
# - Ustaw DNS: A record → IP serwera
# - Zdobądź OpenAI API key

# 2. Wdrożenie (30 minut)
git clone https://github.com/AndyPearman89/PearBlog-Engine-.git
cd PearBlog-Engine-
./scripts/deploy-pt24-pro.sh
./scripts/deploy-pt24-local-services.sh

# 3. Konfiguracja (5 minut)
# - Dodaj OpenAI API key do wp-config.php
# - Zaloguj się do WordPress Admin
# - Sprawdź Enterprise V8 dashboard

# 4. Generowanie treści (1-4 godziny)
./pt24-generate-bulk.sh pages.csv

```
